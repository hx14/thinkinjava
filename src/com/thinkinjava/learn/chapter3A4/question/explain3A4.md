* java的左移与右移
```
 private static int numberTest(int i) {
        if (i == 0) return 32;
        int n = 1;
        // 当i = 2时，2的原二进制转化为：0010，向右移动16/24/28/30位则应得到0，为满足的位数用0补齐
        // 当i = 4时，2的原二进制转化为：0100，向右移动16/24/28/30位则应得到0，为满足的位数用0补齐
        System.out.println(i>>>16);
        System.out.println(i>>>24);
        System.out.println(i>>>28);
        System.out.println(i>>>30);
        
        // 向左移动，注意，满足最大位数32，如果超出就直接跳出循环。
        if (i >>> 16 == 0) {
            n += 16;
            i <<= 16;
        // 当i = 2时，2的原二进制转化为：0010，向左移动之后16位之后得到0010 0000 0000 0000 0000，编译之后省略前面的0，得到10 0000 0000 0000 0000，长度18
        // 当i = 4时，2的原二进制转化为：0100，向左移动之后16位之后得到0100 0000 0000 0000 0000，编译之后省略前面的0，得到100 0000 0000 0000 0000，长度19
            System.out.println(Integer.toBinaryString(i));
        }
        if (i >>> 24 == 0) {
            n += 8;
            i <<= 8;
        // 当i = 2时，2的原二进制转化为：0010，向左移动之后8位之后得到0010 0000 0000 0000 0000 0000 0000，编译之后省略前面的0，得到10 0000 0000 0000 0000 0000 0000，长度22
        // 当i = 4时，2的原二进制转化为：0100，向左移动之后8位之后得到0100 0000 0000 0000 0000 0000 0000，编译之后省略前面的0，得到100 0000 0000 0000 0000 0000 0000，长度23
        
            System.out.println(Integer.toBinaryString(i));
        }
        if (i >>> 28 == 0) {
            n += 4;
            i <<= 4;
        // 当i = 2时，2的原二进制转化为：0010，向左移动之后4位之后得到0010 0000 0000 0000 0000 0000 0000 0000，编译之后省略前面的0，得到10 0000 0000 0000 0000 0000 0000 0000，长度26
        // 当i = 4时，2的原二进制转化为：0100，向左移动之后4位之后得到0100 0000 0000 0000 0000 0000 0000 0000，编译之后省略前面的0，得到100 0000 0000 0000 0000 0000 0000 0000，长度27
        
            System.out.println(Integer.toBinaryString(i));
        }
        if (i >>> 30 == 0) {
            n += 2;
            i <<= 2;
        // 当i = 2时，2的原二进制转化为：0010，向左移动之后2位之后得到1000 0000 0000 0000 0000 0000 0000 0000，编译之后省略前面的0，得到10 0000 0000 0000 0000，长度18
        // 当i = 4时，2的原二进制转化为：0100，向左移动之后大于最大位数32，所以跳出循环
            System.out.println(Integer.toBinaryString(i));
        }
        i >>>= 31;
        n -= i;
        // 此处应该是 n -= i >>> 31，为了方便debug，拆分出来
        return n;
    }

* 随机生成字母
```
    Random random = new Random();
    for (int i = 0; i < 100; i++) {
        // 此处的【+'a'】原因是直接转换为ASCII中的十进制表示
        // random.nextInt(26)是取得[0-26)之间的随机数，再加上一个转化为ASCII的'a'的偏移量用来打印(char)c
        int c = random.nextInt(26) + 'a';
        System.out.println((char) c + ", " + c + ": ");
        switch (c) {
            case 'a':
            case 'e':
            case 'i':
            case 'o':
            case 'u':
                System.out.println("vowel");
                break;
            case 'y':
            case 'w':
                System.out.println("Sometimes a vowel");
                break;
            default:
                System.out.println("consonant");
        }
    }
